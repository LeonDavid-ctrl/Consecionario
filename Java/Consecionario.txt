package com.mycompany.consecionariodojjs;

import java.util.Scanner;

public class ConsecionarioDojjs {
    
    String n,p;
    int s,id,op,w=0,pri;
    admin admi;
    Clientes clientes;
    Productos producto;
    Ventas venta;
    Scanner in = new Scanner(System.in);

    public void main(String[] args) {
        Menu();
    }
    
    public void Menu(){
        do{
        System.out.println("------Bienvenido al concesionario Dojjs------");
        System.out.println("Que desea hacer\n1) Registrarse\n2) Ingresar\n3) Salir");
        op = in.nextInt();
            switch(op){
            case 1: PedirDatos();
                Registro();
            break;
            case 2: PedirDatos();
            w = Ingreso();
            OpUsuario(w);
            op = 2;
            break;
            case 3: System.out.println("Saliendo del programa");
            break;
            default: System.out.println("Opcion Incorrecta");
            break;
        }
        }while(op!=3);
    }
    
    
    public void PedirDatos(){
        in.nextLine();
        System.out.println("Ingrese el usuario");
        n = in.nextLine();
        System.out.println("Ingrese la contraseña");
        p = in.nextLine();
    }
    
    public void Registro(){
        clientes.registrarCliente(n, p);
        System.out.println("Registrado con exito");
    }
    
    public int Ingreso(){
        w = 0;
        if(admi.VerificarAdmin(n, p))
            return 1;
        else
            if(clientes.VerificarCliente(n, p))
                return 2;
            else return 0;
    }
    
    public void OpUsuario(int o){
        switch(o){
            case 0: 
            break;
            case 1: OpAdmin();
            break;
            case 2:OpCliente(); 
            break;
        }
    }
    
    public void OpAdmin(){
        do{
        System.out.println("Opciones de Administrador \n1) Agregar productos\n2) Quitar productos\n3) Ver ventas\n4) Añadir un servicio\n5) Quitar un servicio\n6) Salir");
        op = in.nextInt();
        switch(op){
            case 1: in.nextLine();
                System.out.println("Ingrese el nombre del producto que quiere añadir");
                n = in.nextLine();
                System.out.println("Ingrese el precio del producto");
                pri = in.nextInt();
                producto.AñadirProductos(n,pri);
                System.out.println("Ingrese la cantidad que quiere añadir");
                s = in.nextInt();
                producto.AñadirCantidad(s);
            break;
            case 2: System.out.println("Ingrese el ID del producto que quiere quitar");
                producto.MostrarProductos();
                id = in.nextInt();
                producto.QuitarProductos(id);
            break;
            case 4: in.nextLine();
                System.out.println("Ingrese el nombre del servicio que quiere añadir");
                n = in.nextLine();
                System.out.println("Ingrese el precio del servicio");
                pri = in.nextInt();
                producto.AñadirProductos(n,pri);
            break;
            case 5: System.out.println("Ingrese el ID del servicio que quiere quitar");
                producto.MostrarProductos();
                id = in.nextInt();
                producto.QuitarProductos(id);
            break;
            case 3: venta.ConsultarVentas();
            break;
            case 6: System.out.println("Saliendo");
            break;
            default: System.out.println("Opcion Invalida");
            break;
        }
        }while(op!=6);
    }
    
    public void OpCliente(){
        do{
        System.out.println("Opciones de Administrador \n1) Comprar Productos\n2) Ver los Productos\n3) Salir");
        op = in.nextInt();
        switch(op){
            case 1: System.out.println("Seleccione el ID del producto que quiere comprar");
                producto.VerificarProductos();
                id = in.nextInt();
                producto.VerCantidad(id);
                System.out.println("¿Cuantas unidades quiere comprar?");
                s = in.nextInt();
                producto.Comprar(id, s);
                venta.AgregarVenta(id, producto.nameProducto.get(id), producto.precioProducto.get(id),s);
            break;
            case 2: producto.VerificarProductos();
            break;
            case 3: System.out.println("Saliendo");
            break;
            default: System.out.println("Opcion Invalida");
            break;
        }
        }while(op!=3);
    }
}
